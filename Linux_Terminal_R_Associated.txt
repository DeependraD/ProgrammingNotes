# Adding R's cran repository to the sources.list file, which lists servers that apt-get consults to check whether software exists and where it can be downloaded from.
# Using gedit
sudo gedit /etc/apt/sources.list
# Add this line to the file. This works only for the ubuntu trusty version (14.04 LTS)
deb http://cran.rstudio.com/bin/linux/ubuntu trusty/
# For installing latest R-base
sudo apt-get install r-base
